 CREATE TABLE IF NOT EXISTS users (
  `user_id`         INTEGER  PRIMARY KEY AUTO_INCREMENT,
  `user_name` VARCHAR(50) ,
  `gender`        VARCHAR(2),
  `dob` VARCHAR(25)  ,
  `created_by` VARCHAR(50) ,
   `create_date` TIMESTAMP WITH TIME ZONE,
   `updated_by` VARCHAR(50) ,
   `update_date` TIMESTAMP WITH TIME ZONE
);

DROP TABLE PRODUCT;
CREATE TABLE IF NOT EXISTS `Product` (
  `product_id`         INTEGER  PRIMARY KEY AUTO_INCREMENT,
  `product_name` VARCHAR(50),
  `product_code` VARCHAR(50) ,
  `stock_quantity`        INTEGER,
  `created_by` VARCHAR(50),
   `create_date` TIMESTAMP WITH TIME ZONE,
   `updated_by` VARCHAR(50),
   `update_date` TIMESTAMP WITH TIME ZONE
);

 CREATE TABLE IF NOT EXISTS checkout (
   `checkout_id` INTEGER  PRIMARY KEY AUTO_INCREMENT,
   `user_id` long ,
   `created_by` VARCHAR(50) ,
   `create_date` TIME WITH TIME ZONE,
   `updated_by` VARCHAR(50) ,
   `update_date` TIME WITH TIME ZONE
);
DROP table  checkout_detail ;
CREATE TABLE IF NOT EXISTS checkout_detail (
   `checkout_detail_id` INTEGER  PRIMARY KEY AUTO_INCREMENT,
   `checkout_id` long ,
   `product_id` long ,
   `quantity` long ,
   `created_by` VARCHAR(50) ,
   `create_date` TIME WITH TIME ZONE,
   `updated_by` VARCHAR(50) ,
   `update_date` TIME WITH TIME ZONE
);
